```typescript
// 自定义上传
const customUpload = async (options: any) => {
  const { file, onSuccess, onError } = options

  try {
    uploading.value = true

    // 调用图片上传API
    const response = await uploadApi.uploadImage(file)

    // 保存相对路径到表单
    recipeForm.image = response.url  // 保存: "/static/uploads/images/xxx.jpg"

    // 用于预显示，拼接完整URL
    imageUrl.value = uploadApi.getImageUrl(response.filename)

    console.log('图片上传成功，保存路径:', recipeForm.image)
    console.log('预览URL:', imageUrl.value)

    ElMessage.success('图片上传成功!')
    onSuccess(response)
  } catch (error: any) {
    ElMessage.error('图片上传失败: ' + (error.response?.data?.error || '网络错误'))
    onError(error)
  } finally {
    uploading.value = false
  }
}

// 上传前验证
const beforeImageUpload = (file: File) => {
  const isImage = file.type.startsWith('image/')
  const isLt16M = file.size / 1024 / 1024 < 16

  if (!isImage) {
    ElMessage.error('只能上传图片文件!')
    return false
  }
  if (!isLt16M) {
    ElMessage.error('图片大小不能超过16MB!')
    return false
  }
  return true
}
```

**上传流程**：
1. 用户选择图片
2. 前端验证文件类型和大小
3. 调用 `uploadApi.uploadImage()` 上传到后端
4. 后端保存文件并返回相对路径
5. 前端保存路径到 `recipeForm.image`
6. 前端显示预览图

### 5. 后端创建菜谱逻辑

```python
@recipes_bp.route('/', methods=['POST'])
@jwt_required()  # 需要登录认证
def create_recipe():
    try:
        # 1. 获取并解析请求数据
        data = request.get_json()

        # 2. 验证必填字段
        required_fields = ['title', 'ingredients', 'instructions']
        missing_fields = [field for field in required_fields if not data.get(field)]

        if missing_fields:
            return jsonify({
                'message': f'缺少必填字段: {", ".join(missing_fields)}'
            }), 400

        # 3. 获取当前用户ID
        user_id = get_jwt_identity()

        # 4. 创建Recipe对象
        recipe = Recipe(
            title=data['title'],
            description=data.get('description'),
            ingredients=data['ingredients'],
            instructions=data['instructions'],
            prep_time=data.get('prep_time'),
            cook_time=data.get('cook_time'),
            difficulty=data.get('difficulty'),
            servings=data.get('servings'),
            image=data.get('image'),  # 保存图片路径
            user_id=user_id
        )

        # 5. 保存到数据库
        db.session.add(recipe)
        db.session.commit()

        # 6. 返回创建的菜谱数据
        return jsonify(recipe.to_dict()), 201

    except Exception as e:
        db.session.rollback()  # 回滚事务
        return jsonify({'message': f'服务器错误: {str(e)}'}), 500
```

**核心逻辑**：
1. **JWT认证**：`@jwt_required()` 装饰器
2. **数据验证**：检查必填字段
3. **用户关联**：通过 `get_jwt_identity()` 获取用户ID
4. **数据库操作**：使用 SQLAlchemy ORM
5. **事务管理**：`db.session.commit()` 和 `rollback()`

```python
@recipes_bp.route('/<int:recipe_id>', methods=['PUT'])
@jwt_required()
def update_recipe(recipe_id):
    """更新菜谱 - 只有菜谱创建者可以编辑"""
    user_id = get_jwt_identity()
    user = User.query.get(user_id)
    recipe = Recipe.query.get_or_404(recipe_id)

    # 1. 权限检查：只有创建者可以编辑
    if not user.can_edit_recipe(recipe):
        return jsonify({'message': '无权限编辑此菜谱'}), 403

    data = request.get_json()

    # 2. 更新字段（只更新提供的字段）
    if 'title' in data:
        recipe.title = data['title']
    if 'description' in data:
        recipe.description = data['description']
    if 'ingredients' in data:
        recipe.ingredients = data['ingredients']
    if 'instructions' in data:
        recipe.instructions = data['instructions']
    if 'prep_time' in data:
        recipe.prep_time = data['prep_time']
    if 'cook_time' in data:
        recipe.cook_time = data['cook_time']
    if 'difficulty' in data:
        recipe.difficulty = data['difficulty']
    if 'servings' in data:
        recipe.servings = data['servings']
    if 'image' in data:
        recipe.image = data['image']

    # 3. 提交到数据库
    db.session.commit()

    # 4. 返回更新后的数据
    return jsonify(recipe.to_dict(current_user_id=user_id))
```

**权限控制**：
- 检查用户是否为菜谱创建者
- 管理员也不能编辑他人的菜谱
- 返回 403 禁止访问状态码

### 8. 图片上传后端实现

**文件位置**：`backend/app/routes/uploads.py:31-117`

```python
@uploads_bp.route('/upload-image', methods=['POST'])
@jwt_required()
def upload_image():
    """上传图片文件"""
    try:
        # 1. 检查是否有文件
        if 'image' not in request.files:
            return jsonify({'error': '没有文件部分'}), 400

        file = request.files['image']

        # 2. 验证文件名
        if file.filename == '':
            return jsonify({'error': '没有选择文件'}), 400

        # 3. 验证文件类型
        if not allowed_file(file.filename):
            return jsonify({'error': '不支持的文件类型'}), 400

        # 4. 验证文件大小
        file.seek(0, os.SEEK_END)
        file_size = file.tell()
        file.seek(0)

        if file_size > MAX_FILE_SIZE:
            return jsonify({'error': '文件大小超过16MB限制'}), 400

        # 5. 生成唯一文件名
        file_extension = get_file_extension(file.filename)
        unique_filename = f"{uuid.uuid4().hex}.{file_extension}"

        # 6. 获取上传目录并确保存在
        upload_dir = get_upload_dir()
        os.makedirs(upload_dir, exist_ok=True)

        # 7. 保存文件
        file_path = os.path.join(upload_dir, unique_filename)
        file.save(file_path)

        # 8. 验证文件保存成功
        if not os.path.exists(file_path):
            return jsonify({'error': '文件保存失败'}), 500

        # 9. 返回文件访问路径（相对路径）
        file_url = f"/static/uploads/images/{unique_filename}"

        return jsonify({
            'message': '图片上传成功',
            'filename': unique_filename,
            'url': file_url,
            'size': file_size
        }), 201

    except Exception as e:
        return jsonify({'error': f'文件上传失败: {str(e)}'}), 500
```

**上传关键点**：
1. **文件验证**：类型、大小、文件名
2. **唯一文件名**：使用 UUID 避免冲突
3. **目录管理**：自动创建上传目录
4. **路径处理**：返回相对路径而非绝对路径
5. **错误处理**：详细的错误信息返回
